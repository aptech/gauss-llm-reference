  def get_all_fast_checkers() -> list[BaseChecker]:
      return [c for c in _checkers.values() if not c.requires_sphinx]

  **checkers/code_blocks.py:**
  Implements STRC-01 (code block presence) and STRC-02 (code block non-empty).

  ```python
  class CodeBlockChecker(BaseChecker):
      name = "code_blocks"
      requires_sphinx = False

      def check(self, parsed_doc: ParsedDoc, **kwargs) -> list[Finding]:
          findings = []

          # Skip doc types that don't require code blocks
          if parsed_doc.doc_type in (
              DocType.ALPHA_INDEX,
              DocType.INCLUDE_FRAGMENT,
              DocType.GETTING_STARTED,
              DocType.USER_GUIDE,
              DocType.GRAPHICS_GUIDE,
              DocType.OTHER,
          ):
              return findings

          # STRC-01: COMMAND_REF and OPERATOR pages must have at least one code block
          if not parsed_doc.code_blocks:
              findings.append(Finding(
                  file=parsed_doc.path,
                  line=None,
                  severity=Severity.WARNING,
                  category="missing_code_block",
                  checker="code_blocks",
                  message="Command Reference page has no code blocks",
              ))

          # STRC-02: No empty code blocks (applies to all checked types)
          for block in parsed_doc.code_blocks:
              if block.is_empty:
                  findings.append(Finding(
                      file=parsed_doc.path,
                      line=block.line_number,
                      severity=Severity.ERROR,
                      category="empty_code_block",
                      checker="code_blocks",
                      message="Code block is empty or whitespace-only",
                  ))

          return findings

  register_checker(CodeBlockChecker())
  ```

  **checkers/sections.py:**
  Implements STRC-05 (section structure validation by doc type).

  Required sections by doc type (compare using lowercase, stripped section titles from ParsedDoc.sections):

  | DocType | Required Sections (lowercase) |
  |---------|-------------------------------|
  | COMMAND_REF | "purpose", "format", "examples" |
  | OPERATOR | "purpose", "format", "examples" |
  | APP_MODULE | "purpose", "format", "examples" |
  | All others | No required sections (skip check) |

  ```python
  class SectionChecker(BaseChecker):
      name = "sections"
      requires_sphinx = False

      REQUIRED_SECTIONS = {
          DocType.COMMAND_REF: ["purpose", "format", "examples"],
          DocType.OPERATOR: ["purpose", "format", "examples"],
          DocType.APP_MODULE: ["purpose", "format", "examples"],
      }

      def check(self, parsed_doc: ParsedDoc, **kwargs) -> list[Finding]:
          findings = []
          required = self.REQUIRED_SECTIONS.get(parsed_doc.doc_type)
          if required is None:
              return findings

          present = set(parsed_doc.sections)
          for section in required:
              if section not in present:
                  findings.append(Finding(
                      file=parsed_doc.path,
                      line=None,
                      severity=Severity.WARNING,
                      category="missing_section",
                      checker="sections",
                      message=f"Missing required section: {section.title()}",
                  ))
          return findings

  register_checker(SectionChecker())
  ```

  **checkers/signatures.py:**
  Implements STRC-06 (function signature completeness).

  Only applies to COMMAND_REF and APP_MODULE pages. Skip OPERATOR pages (they use Parameters/Returns sections instead of inline :param: fields).

  Logic:
  1. Look for a `.. function::` directive in the raw_doc. In docutils standalone parsing, `.. function::` is an unknown directive, so it may appear as a `system_message` node or generic directive. Instead of trying to find the directive node itself, use a simpler approach:
     - Search the RST source text (available via `parsed_doc.raw_doc.rawsource` or by re-reading) for lines matching `.. function:: ` pattern
     - Extract the signature from the line after `.. function:: ` (e.g., `y = abs(x)`)
     - Parse the signature to find if it has arguments in parentheses
  2. If the signature has arguments (text between `(` and `)` that is not empty), check that `parsed_doc.field_lists` contains at least one entry with name starting with `param `
  3. If arguments exist but no `:param:` entries found, produce a WARNING finding with category `incomplete_signature`
  4. If no `:return` or `:rtype` entry found in field_lists, produce an INFO finding with category `missing_return_type`
  5. Functions with no arguments (empty parens or no parens) should NOT be flagged for missing params -- this covers cases like `__FILE_DIR` which is a keyword

  **Alternative approach if rawsource is not available:** Walk `parsed_doc.raw_doc.traverse()` looking for `nodes.field_list` nodes. If doc_type is COMMAND_REF and there are NO field_list entries with names starting "param" AND the title suggests it is a function (contains parentheses), flag it.

  ```python
  class SignatureChecker(BaseChecker):
      name = "signatures"
      requires_sphinx = False

      def check(self, parsed_doc: ParsedDoc, **kwargs) -> list[Finding]:
          findings = []

          if parsed_doc.doc_type not in (DocType.COMMAND_REF, DocType.APP_MODULE):
              return findings

          has_params_in_sig = self._signature_has_args(parsed_doc)
          has_param_fields = any(
              f["name"].startswith("param ") for f in parsed_doc.field_lists
          )
          has_return_fields = any(
              f["name"].startswith("return") or f["name"].startswith("rtype")
              for f in parsed_doc.field_lists
          )

          if has_params_in_sig and not has_param_fields:
              findings.append(Finding(
                  file=parsed_doc.path,
                  line=None,
                  severity=Severity.WARNING,
                  category="incomplete_signature",
                  checker="signatures",
                  message="Function has arguments but no :param: documentation",
              ))

          if has_param_fields and not has_return_fields:
              findings.append(Finding(
                  file=parsed_doc.path,
                  line=None,
                  severity=Severity.INFO,
                  category="missing_return_type",
                  checker="signatures",
                  message="Function has :param: fields but no :return: or :rtype: documentation",
              ))

          return findings

      def _signature_has_args(self, parsed_doc: ParsedDoc) -> bool:
          # Check raw source for .. function:: directive with arguments
          import re
          source = parsed_doc.raw_doc.pformat() if hasattr(parsed_doc.raw_doc, 'pformat') else ""
          # Fallback: check the original RST source if stored
          # Look for function-like patterns in the title or field context
          # The most reliable approach: check if field_lists exist at all,
          # implying a function directive was present
          # But we need to distinguish "has args" vs "no args"
          #
          # Strategy: re-read the source from the raw_doc settings
          try:
              source_path = parsed_doc.raw_doc.settings.env.docname if hasattr(parsed_doc.raw_doc.settings, 'env') else None
          except:
              pass

          # Simplest reliable approach: if the title contains parentheses with content
          title = parsed_doc.title
          match = re.search(r'\((.+)\)', title)
          if match:
              return True

          # Also check if any field_list exists (indicating a function directive was present)
          # but only flag if there are args to document
          return False

  register_checker(SignatureChecker())
  ```

  **IMPORTANT NOTE on signature detection:** The above is a starting approach. The executor should test this empirically against the fixture files and the real abs.rst. The key insight is that docutils standalone parsing of `.. function:: y = abs(x)` may put the signature in different places. Test with actual parsing output and adjust. The critical behavior is:

  - incomplete_sig.rst (has `y = myfunc(x, y)` but no `:param:`) -> WARNING incomplete_signature
  - `_file_dir.rst` (no args, no parens) -> no findings (skip)
